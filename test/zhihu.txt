import torch

def compute_grpo_loss(
        log_probs,      # 当前模型的 log_probability
        log_probs_old,  # 旧模型的 log_probability
        advantages,     # 计算好的优势函数 (GRPO 中通常是组内归一化后的奖励)
        eps_clip=0.2    # 建议使用统一的 eps，或者保留你的 eps_low/high
    ):
    # 1. 在 log 空间计算比例，防止数值溢出
    ratio = torch.exp(log_probs - log_probs_old)
    
    # 2. 计算两个代理目标
    surr1 = ratio * advantages
    surr2 = torch.clamp(ratio, 1 - eps_clip, 1 + eps_clip) * advantages
    
    # 3. PPO/GRPO 的核心：取最小收益
    # 注意：如果是为了最大化奖励，Loss 应该是目标的负数
    loss = -torch.min(surr1, surr2).mean() 
    
    return loss

import torch
import torch.nn.functional as F

def compute_gspo_loss(
    log_probs,       # 当前模型的 log_probs, 形状: (B, T)
    log_probs_old,   # 旧模型的 log_probs, 形状: (B, T)
    adv,             # 序列级优势函数, 形状: (B,) 或 (B, 1)
    eps_clip=0.2,    # PPO 剪切范围
    mask=None        # 用于排除 padding token 的掩码
):
    """
    GSPO (Group Sequence Policy Optimization) 损失函数实现
    """
    # 1. 计算每个 token 的对数概率差
    log_ratio = log_probs - log_probs_old
    
    # 2. 序列级聚合与长度归一化 (核心步骤)
    # 如果有 mask，只计算有效 token 的平均值
    if mask is not None:
        log_ratio = log_ratio * mask
        # 几何平均在对数空间表现为有效 token 的算术平均
        seq_log_rho = log_ratio.sum(dim=-1) / mask.sum(dim=-1).clamp(min=1)
    else:
        seq_log_rho = log_ratio.mean(dim=-1)
        
    # 3. 转换回概率空间得到序列级的 rho
    rho = torch.exp(seq_log_rho) # 形状: (B,)
    
    # 4. 执行 Clipped Surrogate Objective
    # 确保 adv 的形状与 rho 匹配 (B,)
    adv = adv.squeeze()
    
    surr1 = rho * adv
    surr2 = torch.clamp(rho, 1 - eps_clip, 1 + eps_clip) * adv
    
    # 5. 计算最终 Loss
    # 因为要 maximize 奖励，所以训练时取负号进行 minimize
    loss = -torch.min(surr1, surr2).mean()
    
    return loss

class DPOLoss(nn.Module):
    """
    DPO Loss
    """

    def __init__(self, beta: float, label_smoothing: float = 0.0, ipo: bool = False) -> None:
        super().__init__()
        self.beta = beta
        self.label_smoothing = label_smoothing
        self.ipo = ipo

    def forward(
        self,
        policy_chosen_logps: torch.Tensor,
        policy_rejected_logps: torch.Tensor,
        reference_chosen_logps: torch.Tensor,
        reference_rejected_logps: torch.Tensor,
    ) -> Tuple[torch.Tensor, torch.Tensor, torch.Tensor]:
        pi_logratios = policy_chosen_logps - policy_rejected_logps
        ref_logratios = reference_chosen_logps - reference_rejected_logps
        logits = pi_logratios - ref_logratios

        if self.ipo:
            losses = (logits - 1 / (2 * self.beta)) ** 2  # Eq. 17 of https://arxiv.org/pdf/2310.12036v2.pdf
        else:
            # Eq. 3 https://ericmitchell.ai/cdpo.pdf; label_smoothing=0 gives original DPO (Eq. 7 of https://arxiv.org/pdf/2305.18290.pdf)
            losses = (
                -F.logsigmoid(self.beta * logits) * (1 - self.label_smoothing)
                - F.logsigmoid(-self.beta * logits) * self.label_smoothing
            )

        loss = losses.mean()
        chosen_rewards = self.beta * (policy_chosen_logps - reference_chosen_logps).detach()
        rejected_rewards = self.beta * (policy_rejected_logps - reference_rejected_logps).detach()

        return loss, chosen_rewards, rejected_rewards

import torch
import math


def rotary_position_embedding(
    q: torch.Tensor, k: torch.Tensor
) -> tuple[torch.Tensor, torch.Tensor]:
    """
    实现RoPE（Rotary Position Embedding）
    输入：
        q: Query张量，形状为 [batch_size, seq_len, head_dim]
        k: Key张量，形状为 [batch_size, seq_len, head_dim]
    输出：
        q_rot: 旋转后的Query
        k_rot: 旋转后的Key
    """
    b, seq_len, head_dim = q.shape

    # 计算 theta: theta_i = 10000^(-2*i/head_dim)，i 是维度索引
    # 形状为 [head_dim // 2]
    dim_indices = torch.arange(head_dim // 2)
    theta = 10000 ** (-2 * dim_indices / head_dim)

    # 计算位置索引: [seq_len, 1]
    positions = torch.arange(seq_len).unsqueeze(1)

    # 计算 position * theta: [seq_len, head_dim // 2]
    angles = positions * theta

    # 计算 cos 和 sin: [seq_len, head_dim // 2]
    cos_theta = angles.cos()
    sin_theta = angles.sin()

    # 扩展到 [seq_len, head_dim // 2] -> [batch, seq_len, head_dim // 2]
    cos_theta = cos_theta.unsqueeze(0).expand(b, -1, -1)
    sin_theta = sin_theta.unsqueeze(0).expand(b, -1, -1)

    # 旋转公式：
    # x_rot[2i] = x[2i] * cos - x[2i+1] * sin
    # x_rot[2i+1] = x[2i] * sin + x[2i+1] * cos
    q_even = q[:, :, 0::2]
    q_odd = q[:, :, 1::2]
    k_even = k[:, :, 0::2]
    k_odd = k[:, :, 1::2]

    q_rot_even = q_even * cos_theta - q_odd * sin_theta
    q_rot_odd = q_even * sin_theta + q_odd * cos_theta
    k_rot_even = k_even * cos_theta - k_odd * sin_theta
    k_rot_odd = k_even * sin_theta + k_odd * cos_theta

    # 合并偶数和奇数维度
    q_rot = torch.stack([q_rot_even, q_rot_odd], dim=-1).reshape(b, seq_len, head_dim)
    k_rot = torch.stack([k_rot_even, k_rot_odd], dim=-1).reshape(b, seq_len, head_dim)

    return q_rot, k_rot

import torch.nn as nn
import torch

class SFTLoss(nn.Module):
    def __init__(self) -> None:
        super().__init__()
        self.loss_fn = nn.CrossEntropyLoss(ignore_index=-100)

    def forward(self, logits: torch.Tensor, labels: torch.Tensor) -> torch.Tensor:
        """
        logits shape (B, L, V)
        labels shape (B, L)
        """
        B, L, V = logits.shape

        logits = logits[:, :-1, :].contiguous()
        labels = labels[:, 1:].contiguous()
        loss = self.loss_fn(logits.view(-1, V), labels.view(-1))    

        return loss

MoE 比 dense 更容易出现训推不一致，本质原因是：它引入了“离散条件计算路径”，而 dense 是连续确定性计算图。

(12.1 结构性质，top-k选取的路由突变)：在Top-k策略下（例如Top-2），如果第2名和第3名的Logits非常接近（例如一个0.1001，一个0.0999），微小的扰动会导致模型选择了完全不同的Expert。因为Expert之间的参数是独立的，这种“路由突变”会导致输出结果发生剧烈的阶跃变化。MoE引入了硬路由（Hard Routing）机制，使得模型不再是一个完全连续可微的系统。在决策边界附近的微小数值扰动，会导致Top-k选择发生跳变，进而导致被激活的参数路径完全不同。

(12.2 浮点数的非结合性， all-reduce等通信顺序不同 -> Router的Logits排序不同)

Dense模型的累加误差通常只是让Loss波动一点点，但MoE中，All-to-All 通信（Expert Parallelism）涉及跨设备的大规模规约（Reduction），通信顺序的不同、CUDA Kernel实现的差异（比如Training用PyTorch原声，Inference用TensorRT或vLLM的定制Kernel），都会放大精度误差。而这个误差一旦影响到Router的Logits排序，就会触发上述的“突变”。

虽然所有模型都有精度误差，但MoE的‘Router排序’对精度极度敏感。Dense模型的精度误差是线性的，而MoE中精度误差可能导致离散的路径选择错误，这是一种非线性的误差放大。

常见的缓解MoE训推不一致的办法：

（1）Routing Replay，有分成R2和R3

（2）TIS或者Mask，分成token和seq层面的处理粒度
